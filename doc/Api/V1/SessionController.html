def check_retrait

      render json: {
        status: true,
        message: request[1]
      }
    else
      render json: {
        status: false,
        message: request[1]
      }
    end
end

def signup_authentication
  phone = params[:phone]
  code  = params[:code]

  authenticate = Parametre::Authentication::validate_2fa(phone, code)
  if authenticate[0] == true
    render json: {
      status: :success,
      message: "Authentifi√©"
    }
  else
    render json: {
      status: :failed,
      message: "Echec authentification"
    }
  end
end

def validate_retrait
