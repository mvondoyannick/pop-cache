def verif_user
    phone = params[:phone]

    #on recherche cet utilisateur
    user = Customer.where(phone: phone).first

    if user
        render json: user.as_json(only: [:name, :second_name]), status: :found
    else
        render json: {
            status: :not_found,
            message: 'unknow user'
        }
    end
end
